# Activation Flow - Analysis and Bug Fix

## Current Issues

### 1. Incorrect Android Code
```java
// DeviceActivator.java - Line 162
String challenge = "xiaozhi-activation-" + System.currentTimeMillis();
```

**Problem**: Self-generating the challenge instead of receiving it from the server!

### 2. Missing OTA Config Step
The Android code does **NOT** call the OTA config API before activating → The server does not return activation data!

### 3. Verification Code Not Displayed Correctly
The 6-digit code is **NOT generated by the client**; it is **returned by the server** in the OTA response!

async def stage_3_ota_config(self):
    """Call OTA API to get config"""

    # GET https://r1.phicomm.com/v1/ota?
    #     sn={serial}&mac={mac}&signature={hmac}&client_id={id}

    config_result = await self.ota.fetch_and_update_config()
    response_data = config_result["response_data"]

    # Check activation data in response
    if "activation" in response_data:
        logger.info("Activation info detected - device needs activation")
        self.activation_data = response_data["activation"]

        self.activation_status["server_activated"] = False
    else:
        logger.info("No activation info - device already activated")
        self.activation_status["server_activated"] = True

### Step 3: POST Activation Request
**File**: `device_activator.py` (line 159-341)

```python
async def activate(self) -> bool:
    """Send activation request with server challenge"""

    # Create HMAC signature with server challenge
    signature = self.device_fingerprint.generate_activation_signature(
        self.challenge  # Challenge FROM SERVER, not self-generated!
    )

    # POST /activate
    payload = {
        "sn": serial_number,
        "mac": mac_address,
        "challenge": self.challenge,  # From server
        "signature": signature,
        "client_id": client_id
    }

    response = await self.session.post(activation_url, json=payload)

    if response.status == 200:
        # Success - user has entered the code
        data = await response.json()
        access_token = data.get("access_token")
        self.device_fingerprint.save_access_token(access_token)
        return True

    elif response.status == 202:
        # Waiting - user has not yet entered the code
        await asyncio.sleep(5)
        continue  # Retry

    else:
        # Error
        return False
```

public class DeviceActivator {
    private String serverChallenge;  // From OTA response
    private String verificationCode;  // From OTA response

    // STEP 1: Call OTA config FIRST
    public void startActivation() {
        OTAConfigManager otaManager = new OTAConfigManager();

        otaManager.fetchOTAConfig(deviceFingerprint, new OTACallback() {
            @Override
            public void onSuccess(OTAResponse response) {
                if (response.activation != null) {
                    // Device not yet activated
                    serverChallenge = response.activation.challenge;
                    verificationCode = response.activation.code;

                    // Display code to user
                    showVerificationCode(verificationCode);

                    // Start polling
                    startActivationPolling();
                } else {
                    // Device already activated
                    onActivationSuccess(null);
                }
            }

            @Override
            public void onError(String error) {
                onActivationError(error);
            }
        });
    }

    // STEP 2: Polling with server challenge
    private void startActivationPolling() {
        // Create HMAC with serverChallenge (do NOT self-generate!)
        String signature = deviceFingerprint.generateActivationSignature(
            serverChallenge  // From server
        );

        JSONObject payload = new JSONObject();
        payload.put("sn", deviceFingerprint.getSerialNumber());
        payload.put("mac", deviceFingerprint.getMacAddress());
        payload.put("challenge", serverChallenge);  // From server
        payload.put("signature", signature);
        payload.put("client_id", getClientId());

        // POST /activate
        // If 202: retry after 5s
        // If 200: save token, done
    }
}
